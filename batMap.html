# Create empty plot
fig, ax = plt.subplots(1,subplot_kw={'aspect':'equal'})

merged.to_crs(epsg=4269).plot(column='Tadarida brasiliensis', scheme='fisher_jenks',
           cmap='GnBu', alpha=.7,
           edgecolor='grey', linewidth=0.8,
           legend=True, legend_kwds={'title' : 'Cases','frameon' : False}, ax=ax)

for index, row in merged.iterrows():
    if index != 'San Francisco' and index != 'San Mateo' :
        plt.annotate(s=index, xy=row['coords'],horizontalalignment='center')
    elif index == 'San Francisco' or index == 'San Mateo' :
        plt.annotate(s=index, xy=row['coords'],horizontalalignment='right')

# Title For Plot
plt.title('Rabies Positive Tadarida brasiliensis By County 2018', fontsize=20)
ax.set_axis_off()

# Save to png
outdir = './data/output/'
runDate = datetime.datetime.now()
fig.savefig(outdir + 'TadaridaBrasiliensisPlot' + runDate.strftime("%Y%m%d") + '.png')
